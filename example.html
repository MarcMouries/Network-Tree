Here is an example on how to use NetworkChart

        var tree_data =
        {
            name: 'root',
            children: [
                { name: 'A' },
                { name: 'B' },
                { name: 'C' },
                { name: 'D' },
            ]
        };

        window.onload = function () {
            NetworkChart.Display("chart", tree_data);
        };
